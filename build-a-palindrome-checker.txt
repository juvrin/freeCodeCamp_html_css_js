** start of script.js **

const inputText = document.getElementById("text-input");
const checkBtn = document.getElementById("check-btn");
const result = document.getElementById("result");

checkBtn.addEventListener("click",() => palindromeChecker(inputText.value));

function palindromeChecker(str){
  if(str === ""){alert("Please input a value")}
  const regex = /[!"#$%&'()*+,-./:;<=>?@[\]^_`{|}~\s]/g;
  let inString = str.replace(regex,"").toLowerCase();
  let revString = inString.split("").reverse().join("");

  let test = inString === revString ? result.textContent =`${str} is a palindrome` : result.textContent =`${str} is not a palindrome`


}
